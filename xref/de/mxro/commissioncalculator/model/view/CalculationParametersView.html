3   import java.math.BigDecimal;
4   
5   import de.mxro.commissioncalculator.model.domain.CalculationParameters;
6   
7   /**
8    * <p>
9    * The parameters required to perform a commission calculation. As used by the
10   * view.
11   * <p>
12   * This class is defined separately to {@link CalculationParameters} to assure
13   * independence of the view layer.
14   * 
15   * @author Max
16   *
17   */
18  public class CalculationParametersView {
19  
20  	/**
21  	 * The target number of sales expected of an agent.
22  	 */
23  	private int target;
24  
25  	/**
26  	 * The actual number of sales an agent performed.
27  	 */
28  	private int actual;
29  
30  	/**
31  	 * The amount the sales person gets paid if the target is met (Monthly On-Target
32  	 * Commission).
33  	 */
34  	private BigDecimal motc;
35  
36  	public int getTarget() {
37  		return target;
38  	}
39  
40  	public void setTarget(int target) {
41  		this.target = target;
42  	}
43  
44  	public int getActual() {
45  		return actual;
46  	}
47  
48  	public void setActual(int actual) {
49  		this.actual = actual;
50  	}
51  
52  	public BigDecimal getMotc() {
53  		return motc;
54  	}
55  
56  	public void setMotc(BigDecimal motc) {
57  		this.motc = motc;
58  	}
59  
60  	@Override
61  	public String toString() {
62  		return "CalculationParametersView [target=" + target + ", actual=" + actual + ", motc=" + motc + "]";
63  	}
64  	
65  	public static CalculationParametersView create(int actual, int target, double motc) {
66  		CalculationParametersView request = new CalculationParametersView();
67  		request.actual = actual;
68  		request.target = target;
69  		request.motc = BigDecimal.valueOf(motc);
70  		return request;
71  	}
72  
73  
74  }
