20  @Service("commissionCalculationService")
21  public class CommissionCalculationServiceImpl implements CommissionCalculationService {
22  
23  	RangeDao range;
24  
25  	@Override
26  	public Commission calculateCommission(CalculationParameters request) {
27  			
28  		// assure here that int is cast to double for the calculation.
29  		double achievement = (double) request.getActual() / (double) request.getTarget();
30  		
31  		Range matchingRange = range.findRangeByAchievement(achievement);
32  		
33  		double base = matchingRange.getBase().doubleValue();
34  		double start = matchingRange.getStart().doubleValue();
35  		double rate = matchingRange.getRate().doubleValue();
36  		
37  		Double commissionRate = base
38  				+ ((achievement - start)
39  						* rate);
40  		
41  		Commission commission = new Commission();
42  
43  		// two digits after decimal point for result
44  		int scale = 2;
45  		commission.setValue(BigDecimal.valueOf(request.getMotc().doubleValue() * commissionRate.doubleValue())
46  				.setScale(scale, BigDecimal.ROUND_HALF_UP));
47  
48  		return commission;
49  	}
50  	
51  	@Autowired
52  	public CommissionCalculationServiceImpl(@Qualifier("hsql") RangeDao range) {
53  		super();
54  		this.range = range;
55  	}
