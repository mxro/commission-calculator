1   package de.mxro.commissioncalculator.db;

3   import javax.sql.DataSource;
4   
5   import org.springframework.context.annotation.Bean;
6   import org.springframework.context.annotation.Configuration;
7   import org.springframework.jdbc.datasource.embedded.EmbeddedDatabase;
8   import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseBuilder;
9   import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseType;
10  
11  /**
12   * <p>A simple Hyper SQL embedded data source used for this project
13   * <p>Data is initialized during application startup by applying the SQL specified
14   * in src/main/resources/sql/create-db.sql and src/main/resources/sql/insert-data.sql
15   * 
16   * @author Max
17   *
18   */
19  @Configuration
20  public class HSQLDataSource {
21  
22  	@Bean
23  	public DataSource dataSource() {
24  		EmbeddedDatabaseBuilder builder = new EmbeddedDatabaseBuilder();
25  		EmbeddedDatabase db = builder.setType(EmbeddedDatabaseType.HSQL).addScript("sql/create-db.sql")
26  				.addScript("sql/insert-data.sql").build();
27  		return db;
28  	}
29  
30  }
